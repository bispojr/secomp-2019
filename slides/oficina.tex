\begin{frame}{Conceito de Proposição}
\begin{block}{Proposição}
	Todo o conjunto de palavras ou símbolos que exprimem um pensamento de sentido completo.
\end{block} \pause
\begin{block}{Características das Proposições}
	\begin{itemize}[label=-] \pause
		\item Afirmam fatos; e  \pause
		\item Exprimem juízos.
	\end{itemize}
\end{block} \pause
\begin{block}{Exemplos}
	\begin{itemize}[label=-]
		\item A Lua é um satélite da Terra.
		\item Recife é a capital de Pernambuco.
		\item $\pi > 5$
		\item sen $\dfrac{\pi}{2} = 1$
	\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Valor Lógico}
\begin{block}{Notação}
	O valor lógico de uma proposição simples $p$ indica-se por $V(p)$.
\end{block} \pause
\begin{block}{Assim...}
	\begin{itemize}[label=-]
		\item Se $p$ é verdadeira, então $V(p) = V$; \pause
		\item Se $p$ é falsa, então $V(p) = F$.
	\end{itemize}
\end{block} \pause
\begin{exampleblock}{Exemplos}
	\begin{itemize}[label=-]
		\item $p$: O Sol é verde.
		\item $q$: Um hexágono tem 9 diagonais.
		\item 2 é raiz da equação $x^2 + 3x -4 = 0$.
	\end{itemize} \pause
	
	\begin{center}
		$V(p) = F$, \ \ \ $V(q) = V$, e \ \ \ $V(r)= F$.
	\end{center}
	
\end{exampleblock}
\end{frame}

\begin{frame}{Questão 3}		
\begin{block}{[Q3]}
Se $V(p) = F$ e $p:$ ``O boné é mais caro do que o chapéu'', então é \underline{\bf impossível} afirmar que...
\end{block}
\begin{enumerate}[label=(\Alph*)]
\item o boné custa o mesmo valor do chapéu.
\item o boné custa R$\$$ $35,00$ e o chapéu custa R$\$$ $40,00$.
\item o boné custa metade do preço do chapéu.
\item o boné custa R$\$$ $15,00$ e o chapéu custa R$\$$ $10,00$.
\end{enumerate}
\end{frame}
